### 베스트 코드

```python
def check(s):
  isPelin = True
  for i in range(len(s)//2):
    if s[i] != s[len(s)-1-i]:
      isPelin = False
      break
  return isPelin

s = list(input())
front = list(s)
back = []
count = 0

for i in range(len(s)):
  if check(s):
    print(len(s)+count)
    break
  else:
    count += 1
    back.insert(0, front[i])
    if check(front+back):
      print(len(s)+count)
      break
```
